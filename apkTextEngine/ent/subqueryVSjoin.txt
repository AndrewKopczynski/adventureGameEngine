-- subquery vs join
--
-- identical (in the sense that they do the same thing)
--
SELECT pub_name
FROM publishers
WHERE pub_id IN
	(SELECT pub_id -- DISTINCT
	FROM titles
	WHERE type = 'biography');
--
--
--
SEELCT pub_name, type
FROM publishers p JOIN titles t
	ON p.pub_id = t.pub_id
WHERE type = 'biography';
